Canvas

<canvas id="myCanvas" height="100" width="100"></canvas> canvas in html
canvas = document.getElementById("myCanvas"); Zeichenfläche schaffen
context = canvas.getContext("2d");

context.fillRect(x, y, breite, höhe); gefülltes Rechteck zeichnen
context.strokeRect(x, y, breite, höhe); Umriss Rechteck zeichnen
context.arc(x, y, radius, startwinkel, endwinkel, anticlockwise); Umriss Kreis zeichnen

context.moveTo(x, y); Position ändern ohne zu zeichnen
context.lineTo(x, y); Linie zeichnen
context.beginPath(); Pfad beginnen
context.endPath(); Pfad schließen
context.stroke(); Pfad einfärben
context.fill(); Fläche im Pfad füllen

canvas.fillStyle = "blue"; Füllfarbe
canvas.strokeStyle = "red"; Strichfarbe
canvas.lineWidth = 2; Strichbreite

canvas.clearRect(x, y, breite, höhe); Löschen von Canvas

canvas.addEventListener("click", function(event) { }); Mausklick-Event
canvas.addEventListener("mousemove", function(event) { }); Mausbewegungs-Event


Beispiel

function drawSymbol(circle) {
    // Zeichnet entweder ein Kreuz oder einen Kreis.
    const canvas = document.getElementById("canvas");
    const context = canvas.getContext('2d');
    context.strokeStyle = 'schwarz';
    context.lineWidth = 5;
    if (circle === true) {
        context.moveTo(25, 75);
        context.arc(75, 75, 50, Math.PI, 3 * Math.PI);
    } else if (circle === false) {
        context.moveTo(25, 25);
        context.lineTo(125, 130);
        context.moveTo(25, 125);
        context.lineTo(125, 25);
    }
    context.stroke();
}
